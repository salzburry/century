# Runtime dependencies for the Century dictionary tooling.
#
# Install once per environment:
#
#     pip install -r requirements.txt
#
# What each dep is for:
#
#   pandas, openpyxl    build_dictionary.py writes Summary / Tables /
#                       Columns / Variables to xlsx (and reads xlsx in
#                       the renderer-omission tests).
#   pyyaml              Both build_dictionary.py and
#                       scripts/validate_packs.py parse
#                       packs/**/*.yaml (cohort + variable + categories
#                       + pii + descriptions).
#   psycopg[binary]     build_dictionary.py opens the warehouse to
#                       actually populate the four pages. Only needed
#                       for live runs; `--dry-run` skips the DB
#                       connection entirely.
#
# Minimum install for offline work
# (pack editing + validator + --dry-run + unit tests):
#
#     pip install pandas openpyxl pyyaml
#
# Add `psycopg[binary]` on top when running against a real cohort.

pandas>=2.0
openpyxl>=3.1
pyyaml>=6.0
psycopg[binary]>=3.1

# Optional. Required only when running
# `python dictionary_v2/discover_exact_matches.py --apply`,
# which round-trips packs/variables/*.yaml to inject `match:` blocks.
# Without this, --apply refuses to write rather than destroy comments.
ruamel.yaml>=0.18
